177  vec_t calcBarycentricCoordinates(const vec_t & p) const {
178  const vec_t ba = b - a;
179  const vec_t ca = c - a;
180  const vec_t cb = c - b;
181  const vec_t ac = a - c;
182  const vec_t pa = p - a;
183  const vec_t pb = p - b;
184  const vec_t pc = p - c;
185 
186  // Normal of triangle (a, b, c)
187  const vec_t n = ba.cross(ca);
188  // Normal of triangle (b, c, p)
189  const vec_t na = cb.cross(pb);
190  // Normal of triangle (c, a, p)
191  const vec_t nb = ac.cross(pc);
192  // Normal of triangle (a, b, p)
193  const vec_t nc = ba.cross(pa);
194 
195  const value_t lengthSquared = n.lengthSquared();
196 
197  return vec_t(n.dot(na) / lengthSquared, n.dot(nb) / lengthSquared, n.dot(nc) / lengthSquared);
198  }

207  vec_t calcCircumcenter() const {
208  const value_t bc2 = getEdgeBC().lengthSquared();
209  const value_t ca2 = getEdgeCA().lengthSquared();

239  vec_t closestPoint(const vec_t & p, _Vec3<value_t> & barycentric) const {
240  // Check if p in vertex region outside a
241  const vec_t ab = b - a;
242  const vec_t ac = c - a;
243  const vec_t ap = p - a;
244  const value_t d1 = ab.dot(ap);
245  const value_t d2 = ac.dot(ap);
246  if (d1 <= 0 && d2 <= 0) {
247  barycentric.setValue(1, 0, 0);
248  return a; // barycentric coordinates (1, 0, 0)
249  }
250 
251  // Check if p in vertex region outside b
252  const vec_t bp = p - b;
253  const value_t d3 = ab.dot(bp);
254  const value_t d4 = ac.dot(bp);
255  if (d3 >= 0 && d4 <= d3) {
256  barycentric.setValue(0, 1, 0);
257  return b; // barycentric coordinates (0, 1, 0)
258  }
259 
260  // Check if p in edge region of ab, if so return projection of p onto ab
261  const value_t vc = d1 * d4 - d3 * d2;
262  if (vc <= 0 && d1 >= 0 && d3 <= 0) {
263  const value_t v = d1 / (d1 - d3);
264  barycentric.setValue(1 - v, v, 0);
265  return a + ab * v; // barycentric coordinates (1 - v, v, 0)
266  }
267 
268  // Check if p in vertex region outside c
269  const vec_t cp = p - c;
270  const value_t d5 = ab.dot(cp);
271  const value_t d6 = ac.dot(cp);
272  if (d6 >= 0 && d5 <= d6) {
273  barycentric.setValue(0, 0, 1);
274  return c; // barycentric coordinates (0, 0, 1)
275  }
276 
277  // Check if p in edge region of ac, if so return projection of p onto ac
278  const value_t vb = d5 * d2 - d1 * d6;
279  if (vb <= 0 && d2 >= 0 && d6 <= 0) {
280  const value_t w = d2 / (d2 - d6);
281  barycentric.setValue(1 - w, 0, w);
282  return a + ac * w; // barycentric coordinates (1 - w, 0, w)
283  }
284 
285  // Check if p in edge region of bc, if so return projection of p onto bc
286  const value_t va = d3 * d6 - d5 * d4;
287  if (va <= 0 && (d4 - d3) >= 0 && (d5 - d6) >= 0) {
288  const value_t w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
289  barycentric.setValue(0, 1 - w, w);
290  return b + (c - b) * w; // barycentric coordinates (0, 1 - w, w)
291  }
292 
293  // p inside face region. Compute result through its barycentric coordinates (u, v, w)
294  const value_t denom = 1 / (va + vb + vc);
295  const value_t v = vb * denom;
296  const value_t w = vc * denom;
297  barycentric.setValue(1 - v - w, v, w);
298  return a + ab * v + ac * w; // = u * a + v * b + w * c, u = va * denom = 1.0f - v - w
299  }

329  bool isDegenerate() const {
330  const value_t minimumArea = std::numeric_limits<value_t>::epsilon();
331  // area = 1/2 * |AB x AC|
332  // area^2 = 1/4 * |AB x AC|^2
333  // 4 * area^2 = |AB x AC|^2
334  return (4.0 * minimumArea * minimumArea) > getEdgeAB().cross(getEdgeAC()).lengthSquared();
335  }
