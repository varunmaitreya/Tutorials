366 template <typename value_t>
367 inline value_t _Box<value_t>::getDistanceSquared(const vec3_t & p) const {
368  value_t x = 0;
369  if (p.x() < min.getX()) {
370  x = min.getX() - p.x();
371  } else if (p.x() > max.getX()) {
372  x = max.getX() - p.x();
373  }
374  value_t y = 0;
375  if (p.y() < min.getY()) {
376  y = min.getY() - p.y();
377  } else if (p.y() > max.getY()) {
378  y = max.getY() - p.y();
379  }
380  value_t z = 0;
381  if (p.z() < min.getZ()) {
382  z = min.getZ() - p.z();
383  } else if (p.z() > max.getZ()) {
384  z = max.getZ() - p.z();
385  }
386  return x * x + y * y + z * z;
387 }
