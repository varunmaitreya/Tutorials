40 template <typename value_t, typename line_t>
41 bool getLineTriangleIntersection(const line_t & line, const Triangle<_Vec3<value_t>> & triangle, value_t & tLine,
42  value_t & uTri, value_t & vTri) {
43  const auto epsilon = std::numeric_limits<value_t>::epsilon();
44 
45  const auto & ori = line.getOrigin();
46  const auto & dir = line.getDirection();
47 
48  // Find vectors for two edges sharing vertex A
49  const auto edgeAB = triangle.getEdgeAB();
50  const auto edgeAC = triangle.getEdgeAC();
51 
52  // Begin calculating determinant - also used to calculate u parameter
53  const auto pVec = dir.cross(edgeAC);
54 
55  // If determinant is near zero, line lies in plane of triangle
56  const value_t det = edgeAB.dot(pVec);
57  if (det > -epsilon && det < epsilon) {
58  return false;
59  }
60  const auto invDet = 1.0 / det;
61 
62  // Calculate distance from vertex A to line origin
63  const auto tVec = ori - triangle.getVertexA();
64 
65  // Calculate u parameter and test bounds
66  const auto u = tVec.dot(pVec) * invDet;
67  if (u < 0.0 || u > 1.0) {
68  return false;
69  }
70 
71  // Prepare to test v parameter
72  const auto qVec = tVec.cross(edgeAB);
73 
74  // Calculate v parameter and test bounds
75  const auto v = dir.dot(qVec) * invDet;
76  if (v < 0.0 || u + v > 1.0) {
77  return false;
78  }
79 
80  // Line intersects triangle: Calculate parameters
81  tLine = edgeAC.dot(qVec) * invDet;
82  uTri = u;
83  vTri = v;
84 
85  return true;
86 }
87 }
