202  /* source:
203  * http://isg.cs.tcd.ie/kavanl/dq/dqconv.c
204  * Converts a unit quaternion and a translation vector into one dual quaternion.
205  */
206  void set(const Quaternion & q, const vec3_t & t) {
207  // regular quaternion (just copy the non-dual part):
208  rot[0] = q.x();
209  rot[1] = q.y();
210  rot[2] = q.z();
211  rot[3] = q.w();
212 
213  // translation vector:
214  trans[0] = 0.5 * (t.x() * q.w() + t.y() * q.z() - t.z() * q.y());
215  trans[1] = 0.5 * (-t.x() * q.z() + t.y() * q.w() + t.z() * q.x());
216  trans[2] = 0.5 * (t.x() * q.y() - t.y() * q.x() + t.z() * q.w());
217  trans[3] = -0.5 * (t.x() * q.x() + t.y() * q.y() + t.z() * q.z());
218  }

224  void getQuaternionAndVec3(Quaternion & q, vec3_t & t) const {
225  // regular quaternion (just copy the non-dual part):
226  q.set(rot[0], rot[1], rot[2], rot[3]);
227  // translation vector:
228  t.setX(2.0 * (-trans[3] * rot[0] + trans[0] * rot[3] - trans[1] * rot[2] + trans[2] * rot[1]));
229  t.setY(2.0 * (-trans[3] * rot[1] + trans[0] * rot[2] + trans[1] * rot[3] - trans[2] * rot[0]));
230  t.setZ(2.0 * (-trans[3] * rot[2] - trans[0] * rot[1] + trans[1] * rot[0] + trans[2] * rot[3]));
231  }

308  static _DualQuaternion dualQuaternionLinearInterpolation(const _DualQuaternion & dq1, const _DualQuaternion & dq2,
309  const value_t factor, bool takeShortestArc = false) {
310  value_t usedFactor = factor;
311  if (takeShortestArc) {
312  if (dq1.dot(dq2) < 0.0)
313  usedFactor = -factor;
314  }
315 
316  _DualQuaternion idq((dq1 * (1.0 - usedFactor)) + (dq2 * usedFactor));
317 
318  value_t length = idq.length();
319  idq.normalize();
320  return (length != 0) ? idq / idq.length() : _DualQuaternion::identity();
321  }
