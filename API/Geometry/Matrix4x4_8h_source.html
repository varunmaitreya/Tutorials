279  _Matrix4x4<value_t> & rotate(const angle_t & angle, value_t x, value_t y, value_t z) {
280  const value_t d = angle.rad();
281  const value_t s = std::sin(d);
282  const value_t c = std::cos(d);
283  const value_t lengthInv = 1.0 / sqrt(x * x + y * y + z * z);
284  x *= lengthInv, y *= lengthInv, z *= lengthInv;
285 
286  const value_t xy = x * y, yz = y * z, xz = x * z, xs = x * s, ys = y * s, zs = z * s;
287  _Matrix4x4<value_t> rot;
288  rot.m[0] = x * x * (1 - c) + c;
289  rot.m[1] = xy * (1 - c) - zs;
290  rot.m[2] = xz * (1 - c) + ys;
291 
292  rot.m[4] = xy * (1 - c) + zs;
293  rot.m[5] = y * y * (1 - c) + c;
294  rot.m[6] = yz * (1 - c) - xs;
295 
296  rot.m[8] = xz * (1 - c) - ys;
297  rot.m[9] = yz * (1 - c) + xs;
298  rot.m[10] = z * z * (1 - c) + c;
299  return (*this) *= rot;
300  }

466  const vec3_t transformDirection(value_t x, value_t y, value_t z) const {
467  /*
468  * w component of a direction is always zero. Simply ignore w here,
469  * because there is no representation of a homogeneous coordinate
470  * with w=0 in cartesian coordinates. Affine transformations always
471  * lead to w=0 after multiplication with a direction. Overall, this
472  * leads to multiplication with the upper-left 3x3 submatrix.
473  */
474  return vec3_t(m[0] * x + m[1] * y + m[2] * z, m[4] * x + m[5] * y + m[6] * z, m[8] * x + m[9] * y + m[10] * z);
475  }

511  value_t d = 1 / det();
