160 template <typename T>
161 bool _Plane<T>::getIntersection(const _Plane<T> & plane, _Line<vec3_t> & intersection) const {
162  // compute direction of intersection line
163  auto dir = plane.getNormal().cross(plane.getNormal());
164  if(dir.lengthSquared() < std::numeric_limits<T>::epsilon())
165  return false;
166  dir.normalize();
167 
168  // compute index to the largest component of D
169  value_t max = std::fabs(dir[0]);
170  uint8_t index = 0;
171  {
172  value_t b = std::fabs(dir[1]);
173  value_t c = std::fabs(dir[2]);
174  if (b > max)
175  max = b, index = 1;
176  if (c > max)
177  max = c, index = 2;
178  }
179 
180  vec3_t p;
181  T d1 = -offset;
182  T d2 = -plane.offset;
183  switch (index) {
184  case 0:
185  p.setValue(0, (d2*normal.z() - d1*plane.normal.z()) / dir.x(), (d1*plane.normal.y() - d2*normal.y()) / dir.x()); break;
186  case 1:
187  p.setValue((d1*plane.normal.z() - d2*normal.z()) / dir.y(), 0, (d2*normal.x() - d1*plane.normal.x()) / dir.y()); break;
188  default:
189  p.setValue((d2*normal.y() - d1*plane.normal.y()) / dir.z(), (d1*plane.normal.x() - d2*normal.x()) / dir.z(), 0);
190  }
191  intersection.setOrigin(p);
192  intersection.setDirection(dir);
193 
194  return true;
195 }
